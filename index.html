# 快速开始

# 安装组件库

npm i crmdemo-ui

# 引用组件库

在 main.js 中引用组件库

//全部引用
import "crmdemo-ui/dist/css/index.css";
import CUI from "crmdemo-ui";
Vue.use(CUI);

//按需引入
import "crmdemo-ui/dist/css/crmDetail.css";
import crmDetail from "crmdemo-ui";
Vue.use(crmDetail);
